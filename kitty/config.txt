include current-theme.conf
# end_kitty_theme

# --- FONTS ---
# begin_kitty_fonts
font_size 15.0
font_family      family='MesloLGL Nerd Font Mono' postscript_name=MesloLGLNFM-Regular
bold_font        auto
italic_font      auto
bold_italic_font auto
# end_kitty_fonts

# --- CURSOR ---
cursor_shape block
cursor_blink_interval 0
cursor_stop_blinking_after 0

# Thickness and Modern UI tweaks
cursor_beam_thickness 1.5
cursor_underline_thickness 2.0
cursor_trail 1
cursor_trail_decay 0.1 0.3

# When window loses focus

# --- MOUSE SETTINGS ---
# This ensures that mouse clicks are passed to Helix even when switching tabs
# and prevents Kitty from "stealing" the click for terminal selection.
mouse_map left click grabbed,ungrabbed mouse_handle_click selection link prompt

# This helps MacOS reporting stay synced
focus_follows_mouse yes
shell_integration enabled

# --- KEYBINDINGS ---
# Command + Enter: new window in current directory
map kitty_mod+enter new_window_with_cwd

map cmd+3 goto_tab 3
map cmd+4 goto_tab 4
map cmd+5 goto_tab 5
map cmd+6 goto_tab 6
map cmd+7 goto_tab 7
map cmd+8 goto_tab 8
map cmd+9 goto_tab 9

# Browse session files
map f6>p goto_session ~/.config/kitty

# --- MACOS SPECIFIC ---
macos_traditional_fullscreen yes
macos_option_as_alt yes
